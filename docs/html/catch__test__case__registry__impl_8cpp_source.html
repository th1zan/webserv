24 struct TestHasher {
25 using hash_t = uint64_t;
26
27 explicit TestHasher( hash_t hashSuffix ):
28 m_hashSuffix{ hashSuffix } {}
29
30 uint32_t operator()( TestCase const& t ) const {
31 // FNV-1a hash with multiplication fold.
32 const hash_t prime = 1099511628211u;
33 hash_t hash = 14695981039346656037u;
34 for ( const char c : t.name ) {
35 hash ^= c;
36 hash *= prime;
37 }
38 hash ^= m_hashSuffix;
39 hash *= prime;
40 const uint32_t low{ static_cast<uint32_t>( hash ) };
41 const uint32_t high{ static_cast<uint32_t>( hash >> 32 ) };
42 return low * high;
43 }
44
45 private:
46 hash_t m_hashSuffix;
47 };
48 } // end unnamed namespace

51 std::vector<TestCase> sortTests( IConfig const& config, std::vector<TestCase> const& unsortedTestCases ) {
52 switch( config.runOrder() ) {
53 case RunTests::InDeclarationOrder:
54 // already in declaration order
55 break;
56
57 case RunTests::InLexicographicalOrder: {
58 std::vector<TestCase> sorted = unsortedTestCases;
59 std::sort( sorted.begin(), sorted.end() );
60 return sorted;
61 }
62
63 case RunTests::InRandomOrder: {
64 seedRng( config );
65 TestHasher h{ config.rngSeed() };
66
67 using hashedTest = std::pair<TestHasher::hash_t, TestCase const*>;
68 std::vector<hashedTest> indexed_tests;
69 indexed_tests.reserve( unsortedTestCases.size() );
70
71 for (auto const& testCase : unsortedTestCases) {
72 indexed_tests.emplace_back(h(testCase), &testCase);
73 }
74
75 std::sort(indexed_tests.begin(), indexed_tests.end(),
76 [](hashedTest const& lhs, hashedTest const& rhs) {
77 if (lhs.first == rhs.first) {
78 return lhs.second->name < rhs.second->name;
79 }
80 return lhs.first < rhs.first;
81 });
82
83 std::vector<TestCase> sorted;
84 sorted.reserve( indexed_tests.size() );
85
86 for (auto const& hashed : indexed_tests) {
87 sorted.emplace_back(*hashed.second);
88 }
89
90 return sorted;
91 }
92 }
93 return unsortedTestCases;
94 }

104 void enforceNoDuplicateTestCases( std::vector<TestCase> const& functions ) {
105 std::set<TestCase> seenFunctions;
106 for( auto const& function : functions ) {
107 auto prev = seenFunctions.insert( function );
108 CATCH_ENFORCE( prev.second,
109 "error: TEST_CASE( \"" << function.name << "\" ) already defined.\n"
110 << "\tFirst seen at " << prev.first->getTestCaseInfo().lineInfo << "\n"
111 << "\tRedefined at " << function.getTestCaseInfo().lineInfo );
112 }
113 }

115 std::vector<TestCase> filterTests( std::vector<TestCase> const& testCases, TestSpec const& testSpec, IConfig const& config ) {
116 std::vector<TestCase> filtered;
117 filtered.reserve( testCases.size() );
118 for (auto const& testCase : testCases) {
119 if ((!testSpec.hasFilters() && !testCase.isHidden()) ||
120 (testSpec.hasFilters() && matchTest(testCase, testSpec, config))) {
121 filtered.push_back(testCase);
122 }
123 }
124 return filtered;
125 }

130 void TestRegistry::registerTest( TestCase const& testCase ) {
131 std::string name = testCase.getTestCaseInfo().name;
132 if( name.empty() ) {
133 ReusableStringStream rss;
134 rss << "Anonymous test case " << ++m_unnamedCount;
135 return registerTest( testCase.withName( rss.str() ) );
136 }
137 m_functions.push_back( testCase );
138 }

143 std::vector<TestCase> const& TestRegistry::getAllTestsSorted( IConfig const& config ) const {
144 if( m_sortedFunctions.empty() )
145 enforceNoDuplicateTestCases( m_functions );
146
147 if( m_currentSortOrder != config.runOrder() || m_sortedFunctions.empty() ) {
148 m_sortedFunctions = sortTests( config, m_functions );
149 m_currentSortOrder = config.runOrder();
150 }
151 return m_sortedFunctions;
152 }

163 std::string extractClassName( StringRef const& classOrQualifiedMethodName ) {
164 std::string className(classOrQualifiedMethodName);
165 if( startsWith( className, '&' ) )
166 {
167 std::size_t lastColons = className.rfind( "::" );
168 std::size_t penultimateColons = className.rfind( "::", lastColons-1 );
169 if( penultimateColons == std::string::npos )
170 penultimateColons = 1;
171 className = className.substr( penultimateColons, lastColons-penultimateColons );
172 }
173 return className;
174 }
