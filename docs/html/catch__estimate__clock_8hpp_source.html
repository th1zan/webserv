30 template <typename Clock>
31 std::vector<double> resolution(int k) {
32 std::vector<TimePoint<Clock>> times;
33 times.reserve(k + 1);
34 std::generate_n(std::back_inserter(times), k + 1, now<Clock>{});
35
36 std::vector<double> deltas;
37 deltas.reserve(k);
38 std::transform(std::next(times.begin()), times.end(), times.begin(),
39 std::back_inserter(deltas),
40 [](TimePoint<Clock> a, TimePoint<Clock> b) { return static_cast<double>((a - b).count()); });
41
42 return deltas;
43 }

45 const auto warmup_iterations = 10000;
46 const auto warmup_time = std::chrono::milliseconds(100);

51 const auto clock_cost_estimation_tick_limit = 100000;
52 const auto clock_cost_estimation_time = std::chrono::milliseconds(10);
53 const auto clock_cost_estimation_iterations = 10000;

55 template <typename Clock>
56 int warmup() {
57 return run_for_at_least<Clock>(std::chrono::duration_cast<ClockDuration<Clock>>(warmup_time), warmup_seed, &resolution<Clock>)
58 .iterations;
59 }
60 template <typename Clock>
61 EnvironmentEstimate<FloatDuration<Clock>> estimate_clock_resolution(int iterations) {
62 auto r = run_for_at_least<Clock>(std::chrono::duration_cast<ClockDuration<Clock>>(clock_resolution_estimation_time), iterations, &resolution<Clock>)
63 .result;
64 return {
65 FloatDuration<Clock>(mean(r.begin(), r.end())),
66 classify_outliers(r.begin(), r.end()),
67 };
68 }
69 template <typename Clock>
70 EnvironmentEstimate<FloatDuration<Clock>> estimate_clock_cost(FloatDuration<Clock> resolution) {
71 auto time_limit = (std::min)(
72 resolution * clock_cost_estimation_tick_limit,
73 FloatDuration<Clock>(clock_cost_estimation_time_limit));
74 auto time_clock = [](int k) {
75 return Detail::measure<Clock>([k] {
76 for (int i = 0; i < k; ++i) {
77 volatile auto ignored = Clock::now();
78 (void)ignored;
79 }
80 }).elapsed;
81 };
82 time_clock(1);
83 int iters = clock_cost_estimation_iterations;
84 auto&& r = run_for_at_least<Clock>(std::chrono::duration_cast<ClockDuration<Clock>>(clock_cost_estimation_time), iters, time_clock);
85 std::vector<double> times;
86 int nsamples = static_cast<int>(std::ceil(time_limit / r.elapsed));
87 times.reserve(nsamples);
88 std::generate_n(std::back_inserter(times), nsamples, [time_clock, &r] {
89 return static_cast<double>((time_clock(r.iterations) / r.iterations).count());
90 });
91 return {
92 FloatDuration<Clock>(mean(times.begin(), times.end())),
93 classify_outliers(times.begin(), times.end()),
94 };
95 }

97 template <typename Clock>
98 Environment<FloatDuration<Clock>> measure_environment() {
99 static Environment<FloatDuration<Clock>>* env = nullptr;
100 if (env) {
101 return *env;
102 }
103
104 auto iters = Detail::warmup<Clock>();
105 auto resolution = Detail::estimate_clock_resolution<Clock>(iters);
106 auto cost = Detail::estimate_clock_cost<Clock>(resolution.mean);
107
108 env = new Environment<FloatDuration<Clock>>{ resolution, cost };
109 return *env;
110 }
