238 struct HasBitOperators {
239 int value;
240
241 friend HasBitOperators operator| (HasBitOperators lhs, HasBitOperators rhs) {
242 return { lhs.value | rhs.value };
243 }
244 friend HasBitOperators operator& (HasBitOperators lhs, HasBitOperators rhs) {
245 return { lhs.value & rhs.value };
246 }
247 friend HasBitOperators operator^ (HasBitOperators lhs, HasBitOperators rhs) {
248 return { lhs.value ^ rhs.value };
249 }
250 explicit operator bool() const {
251 return !!value;
252 }
253
254 friend std::ostream& operator<<(std::ostream& out, HasBitOperators val) {
255 out << "Val: " << val.value;
256 return out;
257 }
258 };
