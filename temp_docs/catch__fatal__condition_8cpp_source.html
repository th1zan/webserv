78 // There is no 1-1 mapping between signals and windows exceptions.
79 // Windows can easily distinguish between SO and SigSegV,
80 // but SigInt, SigTerm, etc are handled differently.
81 static SignalDefs signalDefs[] = {
82 { static_cast<DWORD>(EXCEPTION_ILLEGAL_INSTRUCTION), "SIGILL - Illegal instruction signal" },
83 { static_cast<DWORD>(EXCEPTION_STACK_OVERFLOW), "SIGSEGV - Stack overflow" },
84 { static_cast<DWORD>(EXCEPTION_ACCESS_VIOLATION), "SIGSEGV - Segmentation violation signal" },
85 { static_cast<DWORD>(EXCEPTION_INT_DIVIDE_BY_ZERO), "Divide by zero error" },
86 };
87
88 static LONG CALLBACK handleVectoredException(PEXCEPTION_POINTERS ExceptionInfo) {
89 for (auto const& def : signalDefs) {
90 if (ExceptionInfo->ExceptionRecord->ExceptionCode == def.id) {
91 reportFatal(def.name);
92 }
93 }
94 // If its not an exception we care about, pass it along.
95 // This stops us from eating debugger breaks etc.
96 return EXCEPTION_CONTINUE_SEARCH;
97 }

99 // Since we do not support multiple instantiations, we put these
100 // into global variables and rely on cleaning them up in outlined
101 // constructors/destructors
102 static PVOID exceptionHandlerHandle = nullptr;
103
104
105 // For MSVC, we reserve part of the stack memory for handling
106 // memory overflow structured exception.
107 FatalConditionHandler::FatalConditionHandler() {
108 ULONG guaranteeSize = static_cast<ULONG>(minStackSizeForErrors);
109 if (!SetThreadStackGuarantee(&guaranteeSize)) {
110 // We do not want to fully error out, because needing
111 // the stack reserve should be rare enough anyway.
112 Catch::cerr()
113 << "Failed to reserve piece of stack."
114 << " Stack overflows will not be reported successfully.";
115 }
116 }
117
118 // We do not attempt to unset the stack guarantee, because
119 // Windows does not support lowering the stack size guarantee.
120 FatalConditionHandler::~FatalConditionHandler() = default;

123 void FatalConditionHandler::engage_platform() {
124 // Register as first handler in current chain
125 exceptionHandlerHandle = AddVectoredExceptionHandler(1, handleVectoredException);
126 if (!exceptionHandlerHandle) {
127 CATCH_RUNTIME_ERROR("Could not register vectored exception handler");
128 }
129 }
130
131 void FatalConditionHandler::disengage_platform() {
132 if (!RemoveVectoredExceptionHandler(exceptionHandlerHandle)) {
133 CATCH_RUNTIME_ERROR("Could not unregister vectored exception handler");
134 }
135 exceptionHandlerHandle = nullptr;
136 }

156 { SIGFPE, "SIGFPE - Floating point error signal" },
157 { SIGSEGV, "SIGSEGV - Segmentation violation signal" },

162// Older GCCs trigger -Wmissing-field-initializers for T foo = {}
163// which is zero initialization, but not explicit. We want to avoid
164// that.
165#if defined(__GNUC__)
166# pragma GCC diagnostic push
167# pragma GCC diagnostic ignored "-Wmissing-field-initializers"
168#endif
169
170 static char* altStackMem = nullptr;
171 static std::size_t altStackSize = 0;
172 static stack_t oldSigStack{};
173 static struct sigaction oldSigActions[sizeof(signalDefs) / sizeof(SignalDefs)]{};
174
175 static void restorePreviousSignalHandlers() {
176 // We set signal handlers back to the previous ones. Hopefully
177 // nobody overwrote them in the meantime, and doesn't expect
178 // their signal handlers to live past ours given that they
179 // installed them after ours..
180 for (std::size_t i = 0; i < sizeof(signalDefs) / sizeof(SignalDefs); ++i) {
181 sigaction(signalDefs[i].id, &oldSigActions[i], nullptr);
182 }
183 // Return the old stack
184 sigaltstack(&oldSigStack, nullptr);
185 }
186
187 static void handleSignal( int sig ) {
188 char const * name = "<unknown signal>";
189 for (auto const& def : signalDefs) {
190 if (sig == def.id) {
191 name = def.name;
192 break;
193 }
194 }
195 // We need to restore previous signal handlers and let them do
196 // their thing, so that the users can have the debugger break
197 // when a signal is raised, and so on.
198 restorePreviousSignalHandlers();
199 reportFatal( name );
200 raise( sig );
201 }

203 FatalConditionHandler::FatalConditionHandler() {
204 assert(!altStackMem && "Cannot initialize POSIX signal handler when one already exists");
205 if (altStackSize == 0) {
206 altStackSize = std::max(static_cast<size_t>(SIGSTKSZ), minStackSizeForErrors);
207 }
208 altStackMem = new char[altStackSize]();
209 }
210
211 FatalConditionHandler::~FatalConditionHandler() {
212 delete[] altStackMem;
213 // We signal that another instance can be constructed by zeroing
214 // out the pointer.
215 altStackMem = nullptr;
216 }
217
218 void FatalConditionHandler::engage_platform() {
219 stack_t sigStack;
220 sigStack.ss_sp = altStackMem;
221 sigStack.ss_size = altStackSize;
222 sigStack.ss_flags = 0;
223 sigaltstack(&sigStack, &oldSigStack);
224 struct sigaction sa = { };
225
226 sa.sa_handler = handleSignal;
227 sa.sa_flags = SA_ONSTACK;
228 for (std::size_t i = 0; i < sizeof(signalDefs)/sizeof(SignalDefs); ++i) {
229 sigaction(signalDefs[i].id, &sa, &oldSigActions[i]);
230 }
231 }
