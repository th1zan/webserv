24namespace Catch {
25 namespace Benchmark {
26 template <typename Duration>
27 struct ExecutionPlan {
28 int iterations_per_sample;
29 Duration estimated_duration;
30 Detail::BenchmarkFunction benchmark;
31 Duration warmup_time;
32 int warmup_iterations;
33
34 template <typename Duration2>
35 operator ExecutionPlan<Duration2>() const {
36 return { iterations_per_sample, estimated_duration, benchmark, warmup_time, warmup_iterations };
37 }
38
39 template <typename Clock>
40 std::vector<FloatDuration<Clock>> run(const IConfig &cfg, Environment<FloatDuration<Clock>> env) const {
41 // warmup a bit
42 Detail::run_for_at_least<Clock>(std::chrono::duration_cast<ClockDuration<Clock>>(warmup_time), warmup_iterations, Detail::repeat(now<Clock>{}));
43
44 std::vector<FloatDuration<Clock>> times;
45 times.reserve(cfg.benchmarkSamples());
46 std::generate_n(std::back_inserter(times), cfg.benchmarkSamples(), [this, env] {
47 Detail::ChronometerModel<Clock> model;
48 this->benchmark(Chronometer(model, iterations_per_sample));
49 auto sample_time = model.elapsed() - env.clock_cost.mean;
50 if (sample_time < FloatDuration<Clock>::zero()) sample_time = FloatDuration<Clock>::zero();
51 return sample_time / iterations_per_sample;
52 });
53 return times;
54 }
55 };
